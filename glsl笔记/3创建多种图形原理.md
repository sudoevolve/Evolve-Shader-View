设置和控制几何体的形状、位置和大小。

## 形状设置的完整解释：

```glsl
// SDF函数（这些定义了形状的"公式"）
float sphereSDF(vec3 p, vec3 center, float radius)
{
    return length(p - center) - radius;
}

float planeSDF(vec3 p, vec3 normal, float distanceFromOrigin)
{
    return dot(p, normal) - distanceFromOrigin;
}

float boxSDF(vec3 p, vec3 center, vec3 size)
{
    vec3 d = abs(p - center) - size;
    return min(max(d.x, max(d.y, d.z)), 0.0) + length(max(d, 0.0));
}
```

## 如何在场景中使用这些形状：

```glsl
void mainImage(out vec4 fragColor, in vec2 fragCoord)
{
    // ... 相机设置代码 ...
    
    // 在射线步进循环中使用这些形状：
    for(int i = 0; i < 100; i++)
    {
        // 每个物体的参数都在这里设置！
        
        // 1. 红色球体
        float sphereDist = sphereSDF(
            rayPos,                    // 当前射线位置
            vec3(-2.0, 0.0, 0.0),     // 球心位置 (x=-2, y=0, z=0)
            1.0                       // 半径 = 1.0
        );
        
        // 2. 绿色立方体
        float boxDist = boxSDF(
            rayPos,                    // 当前射线位置  
            vec3(2.0, 0.0, 0.0),      // 立方体中心位置 (x=2, y=0, z=0)
            vec3(0.8, 0.8, 0.8)       // 立方体半尺寸 (宽度=1.6, 高度=1.6, 深度=1.6)
        );
        
        // 3. 蓝色平面
        float planeDist = planeSDF(
            rayPos,                    // 当前射线位置
            vec3(0.0, 1.0, 0.0),      // 平面法向量 (朝上)
            -1.0                      // 平面到原点的距离 (y=-1的位置)
        );
        
        // 取最近的距离
        float minDist = min(min(sphereDist, boxDist), planeDist);
        
        // ... 碰撞检测代码 ...
    }
}
```

## 详细参数说明：

### 1. 球体参数设置：
```glsl
sphereSDF(rayPos, vec3(x, y, z), radius)

// 例子：
sphereSDF(rayPos, vec3(-2.0, 0.0, 0.0), 1.0)     // 左边的球
sphereSDF(rayPos, vec3(0.0, 2.0, 0.0), 0.5)      // 上方的小球
sphereSDF(rayPos, vec3(3.0, -1.0, 2.0), 1.5)     // 右后方的大球
```

### 2. 立方体参数设置：
```glsl
boxSDF(rayPos, vec3(x, y, z), vec3(width, height, depth))

// 注意：size参数是"半尺寸"！
boxSDF(rayPos, vec3(2.0, 0.0, 0.0), vec3(0.8, 0.8, 0.8))  // 1.6×1.6×1.6的立方体
boxSDF(rayPos, vec3(0.0, 1.0, 0.0), vec3(2.0, 0.2, 0.5))  // 扁平的长方体
boxSDF(rayPos, vec3(-1.0, 0.0, 1.0), vec3(0.3, 1.0, 0.3)) // 细长的柱子
```

### 3. 平面参数设置：
```glsl
planeSDF(rayPos, vec3(normalX, normalY, normalZ), distance)

// 例子：
planeSDF(rayPos, vec3(0.0, 1.0, 0.0), -1.0)   // 水平地面 (y=-1)
planeSDF(rayPos, vec3(0.0, 1.0, 0.0), 0.0)    // XY平面 (y=0)
planeSDF(rayPos, vec3(1.0, 0.0, 0.0), -2.0)   // 垂直墙面 (x=-2)
planeSDF(rayPos, vec3(0.0, 0.0, 1.0), 1.0)    // 垂直墙面 (z=1)
```

## 完整的可调整示例：

```glsl
float sphereSDF(vec3 p, vec3 center, float radius)
{
    return length(p - center) - radius;
}

void mainImage(out vec4 fragColor, in vec2 fragCoord)
{
    // 相机设置
    vec2 uv = (fragCoord / iResolution.xy) * 2.0 - 1.0;
    uv.x *= iResolution.x / iResolution.y;
    vec3 ro = vec3(0.0, 0.0, -5.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    // === 在这里调整形状参数 ===
    
    // 球体1：位置和大小
    vec3 sphere1Pos = vec3(-1.5, 0.0, 0.0);  // 位置
    float sphere1Radius = 0.8;               // 半径
    
    // 球体2：位置和大小  
    vec3 sphere2Pos = vec3(1.5, 0.5, 0.0);   // 位置
    float sphere2Radius = 0.6;               // 半径
    
    vec3 p = ro;
    bool hit = false;
    int hitSphere = 0;  // 1=球体1, 2=球体2
    
    for(int i = 0; i < 100; i++)
    {
        float d1 = sphereSDF(p, sphere1Pos, sphere1Radius);
        float d2 = sphereSDF(p, sphere2Pos, sphere2Radius);
        float d = min(d1, d2);
        
        if(d < 0.001)
        {
            hit = true;
            hitSphere = (d == d1) ? 1 : 2;
            break;
        }
        
        if(d > 100.0) break;
        p += rd * d;
    }
    
    vec3 color;
    if(hit)
    {
        if(hitSphere == 1) 
            color = vec3(1.0, 0.0, 0.0);  // 红色球体1
        else 
            color = vec3(0.0, 1.0, 0.0);  // 绿色球体2
    }
    else
    {
        color = vec3(0.0, 0.0, 0.0);      // 黑色背景
    }
    
    fragColor = vec4(color, 1.0);
}
```

## 关键要点：

1. **SDF函数只是"公式"** - 它们定义了形状的数学性质
2. **实际的形状参数在调用时设置** - 每次调用时传入不同的参数
3. **可以创建多个相同类型的形状** - 只要使用不同的参数
4. **参数可以随时修改** - 改变位置、大小、旋转等

你可以随意修改这些参数来创建不同的场景！
